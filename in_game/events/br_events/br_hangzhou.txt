namespace = hangzhou

hangzhou.1 = { #Receiving petition to send bishop
	type = country_event
	title = hangzhou.1.title
	desc = hangzhou.1.desc
	
	fire_only_once = yes
	dynamic_historical_event = {
		tag = AOI # Expedition starts from Antioch
		from = 1300.1.1
		to = 1500.1.1
		monthly_chance = 100
	}
	
	option = { #we send bishop
		name = hangzhou.1.a
		historical_option = yes
		add_gold = -1000 # we give money for trip, maybe option in the future for additional money (the more money you invest the more safe the trip)
		add_religious_influence = 100
		hidden_effect = {
			trigger_event_non_silently = { id = hangzhou.2 days = 3 }
		}
	}
	
	option = { #we do not send bishop
		name = hangzhou.1.b
		add_prestige = -25
	}
}

hangzhou.2 = { #Bishop arrives in Hangzhou
	type = country_event
	title = hangzhou.2.title
	desc = hangzhou.2.desc
	
	fire_only_once = yes
	dynamic_historical_event = {
		tag = AOI
	}

	immediate = {
		c:CHI = {
			release_tag = { tag = POH }
		}
		
		c:POH = {
			make_subject_of = {
				target = c:CHI
				type = subject_type:tributary
			}
			add_reform = government_reform:patriarchate
		}
		
		location:yuhang = {
			change_location_owner = c:POH
			remove_core = c:CHI			
		}
		location:fuyang_hangzhou = {
			change_location_owner = c:POH
			remove_core = c:CHI			
		}
		location:xiaoshan = {
			change_location_owner = c:POH
			remove_core = c:CHI		
		}
		location:shaoxing = {
			change_location_owner = c:POH
			remove_core = c:CHI			
		}
		location:shangyu = {
			change_location_owner = c:POH
			remove_core = c:CHI			
		}
		location:haining = {
			change_location_owner = c:POH
			remove_core = c:CHI			
		}
		location:hangzhou = {
			change_location_owner = c:POH
			remove_core = c:CHI			
        }
		location:deqing = {
			change_location_owner = c:POH
			remove_core = c:CHI
		}
		location:chongde = {
			change_location_owner = c:POH
			remove_core = c:CHI
		}
		international_organization:middle_kingdom = {
			add_country_to_international_organization = c:POH
		}
	}

	option = { # stay as Antioch, we wish them luck
		name = hangzhou.2.a
		add_religious_influence = 100
    }
	option = { #change to Patriarchate
		name = hangzhou.2.b
		trigger = { hidden_trigger = { is_ai = no } }
		add_religious_influence = 100
		change_player = c:POH
	}
	after = {
		c:POH = {
			trigger_event_non_silently = { id = hangzhou.3 days = 7 }
			create_character = {
				culture = culture:seres_culture
				religion = religion:orthodox
				estate = estate_type:clergy_estate
				min_age = 25
				save_scope_as = target_character
			}
			set_new_ruler = scope:target_character
			set_variable = {
				name = character_creator
				value = 20	#His 20 buddies
			}
			while = {
				limit = { var:character_creator > 0 }
				change_variable = {
					name = character_creator
					subtract = 1
				}
				create_character = {
					culture = culture:seres_culture
					religion = religion:orthodox
					min_age = 18
				}
			}
			remove_variable = character_creator
			
			hidden_effect = {
				
				#graning privileges to Patriarchate of Hangzhou
				
				#nobility
				grant_estate_privilege = estate_privilege:banal_lordship
				grant_estate_privilege = estate_privilege:br_dynatoi_magnates
				grant_estate_privilege = estate_privilege:monetary_fiefs
				grant_estate_privilege = estate_privilege:noble_serfdom_rights
				
				#clergy
				grant_estate_privilege = estate_privilege:clergy_in_administration
				grant_estate_privilege = estate_privilege:br_bureaucrat_clergy
				grant_estate_privilege = estate_privilege:clergy_powerful_dioceses
				grant_estate_privilege = estate_privilege:clergy_religious_state
				grant_estate_privilege = estate_privilege:clerical_advisory_council
				grant_estate_privilege = estate_privilege:tax_exemption
				
				#burghers
				grant_estate_privilege = estate_privilege:building_roads_rights
				grant_estate_privilege = estate_privilege:market_fairs
				grant_estate_privilege = estate_privilege:land_of_commerce
				
				#peasants
				grant_estate_privilege = estate_privilege:communal_lands
				grant_estate_privilege = estate_privilege:no_labor_sunday
			
			
				add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_radical_bonus }
				add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_radical_bonus }
				add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_radical_bonus }
				add_estate_satisfaction = { type = estate_type:burghers_estate value = estate_satisfaction_radical_bonus }
			}
		}
		location:hangzhou = {
			add_pop = {
				type = pop_type:clergy
				culture = culture:seres_culture
				religion = religion:orthodox
				size = 0.1
				literacy = 50
			}			
        }
	}
}
