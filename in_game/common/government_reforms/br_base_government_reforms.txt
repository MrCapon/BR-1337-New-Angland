		num_of_ports > 0
		has_reform = government_reform:military_order_reform

		#other countries can access it if they are neightboors of the Knights or
		#if they are its vassal
		OR = {
			has_or_had_tag = KNI
			any_neighbor_country = { has_reform = government_reform:crusade_sea_reform }
			overlord ?= { has_reform = government_reform:crusade_sea_reform }

		}

		global_navy_levy_size_modifier = 0.5
