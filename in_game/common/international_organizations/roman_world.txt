roman_world = {
	has_target = no
	unique = yes
	expel_members_who_are_targets_of_other_members = no
	
	show_leave_message = no
	use_laws_as_join_reason = no
	has_leader_country = yes
	show_on_diplomatic_map = yes
	leader_type = country
	gives_food_access_to_members = yes

	show_as_overlord_on_map_trigger = {
		always = yes
	}

	
	has_leader_country = yes
	leader_title_key = "RW_LEADER"
	use_regnal_number = yes
	leader = {
		leader_country ?= {
			if = {
				limit = {
					has_regent = yes
					has_heir = yes
				}
				heir ?= {
					add_to_list = leaders
				}
			}
			else_if = {
				limit = {
					has_ruler = yes
					ruler ?= { is_alive = yes }
				}
				ruler ?= {
					add_to_list = leaders
				}
			}
		}
	}

	#only_leader_country_joins_defensive_wars = yes

	join_defensive_wars_auto_call = {
		scope:target ?= {
			NOT = {
				is_member_of_international_organization = root
			}
		}
	}
	
	
	can_declare_war = {
		trigger_if = {
			limit = {
				scope:recipient = { modifier:enforced_internal_peace = yes }
				scope:attacker = { is_member_of_international_organization = scope:recipient }
				scope:defender = { is_member_of_international_organization = scope:recipient }
			}
			scope:recipient = { modifier:enforced_internal_peace = no }
		}
		trigger_else = {
			always = yes
		}
	}

	gives_military_access_to_all_when_at_war = yes

	land_ownership_rule = roman_land_ownership
	#antagonism_modifier_for_taking_land_from_fellow_member = -2.0	  # Members get less Antagonism in the ERE
	antagonism_modifier_for_taking_land_from_member_as_outsider = 2.0 # Outsiders get more Antagonism in the ERE
	#no_cb_price_modifier_for_fellow_member = 0.75

	non_leader_modifier = {
		
		#positive modifiers
		fort_maintenance_cost = -0.10
		global_defensive = 0.25
		global_hostile_attrition = 0.10

		#negative modifiers
		block_from_change_to_empire_rank = yes
		diplomatic_capacity_modifier = -0.5
		blocked_from_conversion = yes
	}

	variables = {
		roman_authority = {
			format = "celestial_authority_DISPLAY"
			change_format = "VARIABLE_CHANGE_FORMAT"
			min = 0
			max = 100
			start = 70
			monthly_change = {
				subtract = {
					desc = "BASE_DECAY"
					value = 0.1
				}
				add = {	
					#internal wars
					if = {
						limit = {
							any_international_organization_member = {
								any_country_at_war_with = {
									is_member_of_international_organization = root 
								}
							}
						}
						add = {
							desc = "MIDDLE_KINGDOM_MEMBERS_AT_WAR"
							value = 0
							every_international_organization_member = {
								limit = {
									any_country_at_war_with = {
										is_member_of_international_organization = root 
									}
								}
								add = -0.01
							}
						}
					}
					#Number of Tributaries
					add = {
						desc = "NUMBER_OF_ROMAN_MEMBERS"
						value = 0
						every_international_organization_member = {
							limit = {
								NOT = { is_leader_of_international_organization = root }
							}
							add = {
								value = 0.02
								if = {
									limit = { exists = root.leader_country }
									multiply = {
										value = "opinion(root.leader_country)"
										divide = 100
									}
								}
								multiply = {
									value = great_power_score
									divide = 100
								}
							}
						}
					}
				}
				if = {
					limit = { 
						NOT = { 
							leader_country = { modifier:monthly_roman_authority = 0 }
						}
					}
					add = {
						desc = "IO_MONTHLY_ROMAN_AUTHORITY"
						#value = 0
						value = leader_country.modifier:monthly_roman_authority
					}
				}
			}
		}
	}
	
	leader_modifier = {
		diplomatic_capacity = 2
		max_diplomats = 2
		monthly_diplomats = 0.5
		#subject_loyalty = 10
		monthly_prestige = 0.15
		court_spending_cost = 0.05
		
		allowed_intervene_in_war = yes
		allowed_enforce_peace = yes
		allowed_threaten_war = yes
		allow_tributary_subject = yes
		force_convert_created_subjects = yes


		num_possible_artists = 1 
		government_size = 1
		num_local_governors = 1

	}
	
	create_visible_trigger = {
		always = no
	}

	invite_visible_trigger = {
		always = yes
	}

	can_lead_trigger = {
		OR = {
			has_or_had_tag = BYZ
			AND = {
				religion = religion:orthodox
				country_rank = country_rank:rank_empire
			}
		}
	}
	
	can_join_trigger = {
		scope:recipient = {
			international_organization_has_leader = yes
		}
		religion = scope:recipient.leader_country.religion
		is_neighbor_of_international_organization = scope:recipient
	}
	
	can_leave_trigger = {
		NOT = { is_subject_of = scope:recipient.leader_country }
		NOT = { is_leader_of_international_organization = scope:recipient }
	}
	
	auto_leave_trigger = {
		always = no
	}

	
	auto_disband_trigger = {
		always = no
	}

	on_joined = {
	}

	on_left = {
	}

	# TO-DO add tiered membership types
	#special_statuses_implemented = {
	#}

	#TO-DO add unqiue laws
	#laws = {
	#}
	
	monthly_effect = {
		leader_country ?= {
			every_subject = {
				limit = { NOT = { is_member_of_international_organization = root } }
				root = {
					add_country_to_international_organization = prev
				}
			}				
		}
	}
	
	payments_implemented = {
		roman_tribute
		#imperial_contribution
	}


	ai_desire_to_join = {
		add = {
			desc = "OPINION_OF_ROMAN_EMPEROR"
			value = {
				scope:recipient = {
					leader_country = {
						root = {
							value = this_opinion_of_prev
							multiply = 0.2
						}
					}
				}
			}
		}
		if = {
			limit = {
				opinion = {
					target = scope:recipient.leader_country
					value < -20
				}
			}
			add = {
				desc = "POOR_RELATIONS"
				value = -1000
			}
		}
		add = {
			desc = "EMPEROR_DIP_REPUTATION"
			scope:recipient = {
				value = leader_country.modifier:diplomatic_reputation
				multiply = 2
			}
		}
		if = {
			limit = {
				NOT = { within_diplomatic_range = scope:recipient.leader_country }
			}
			add = {
				desc = "OUTSIDE_OF_DIP_RANGE"
				value = -1000
			}
		}
		add = {
			desc = "DISTANCE_FROM_CHINA"
			value = "border_distance_to(scope:recipient.leader_country)"
			multiply = -0.05
			min = -1000
		}
		if = {
			limit = { at_war = yes }
			add = {
				desc = "IS_AT_WAR"
				value = -1000
			}
		}
	}
	
	ai_desire_to_allow_new_member = {
		add = {
			desc = "WE_WANT_THE_ORGANIZATION_TO_GROW"
			value = 20
		}
		if = {
			limit = {
				leader_country = { is_during_bankruptcy = yes }
			}
			add = {
				desc = "WE_EXPERIENCE_BANKRUPTCY"
				value = -1000
			}
		}
		add = {
			desc = "CURRENT_ROMAN_AUTHORITY"
			root = { 
				value = var:roman_authority 
				multiply = 0.05
			}
		}
		scope:actor = {
			add = {
				desc = "ECONOMIC_SIZE"
				value = monthly_income_trade_and_tax
				multiply = -1
			}
			if = {
				limit = {
					is_subject_of = root.leader_country
				}
				add = {
					desc = "NOT_SUBJECT_OF_EMPEROR"
					value = -25
				}
			}
		}
	}
}
