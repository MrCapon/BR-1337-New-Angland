#leader pays tributes to members in exchange for getting more mandate.
roman_tribute = {
	get_payer_list = {
		every_international_organization_member = {
			limit = { 
				NOT = { is_leader_of_international_organization = root }
			}
			add_to_local_variable_list = {
				name = payers
				target = this
			}
		}
	}
	get_payee_list = {
		leader_country ?= {
			add_to_local_variable_list = {
				name = payees
				target = this
			}
		}
	}
	price = roman_tribute_price
	uses_maintenance = yes
	min_slider_value = 0 
	maintenance_modifier = {
		potential_trigger = {
			"payment_maintenance(scope:recipient|roman_tribute)" < 0.5
		}
		scale = {
			value = 1
			subtract = {
				value = "payment_maintenance(scope:recipient|roman_tribute)"
				multiply = 2
			}
		}
		global_crown_estate_power = -0.2
	}
	maintenance_modifier = {
		potential_trigger = {
			"payment_maintenance(scope:recipient|roman_tribute)" > 0.5
		}
		scale = {
			value = "payment_maintenance(scope:recipient|roman_tribute)"
			subtract = 0.5
			multiply = 2
		}
		global_crown_estate_power = 0.2
	}
	price_multiplier = {
		add = {
			desc = roman_tribute_price_modifier
			every_international_organization_member = {
				limit = { 
					not = { is_leader_of_international_organization = root }
				}
				add = {
					value = monthly_income_trade_and_tax
					multiply = {
						value = 0.25 #1
						add = modifier:tribute_payment_received_modifier
					}
				}
			}
		}
	}
	proportion_for_payer = {
		add = {
			desc = "roman_tribute"
			value = monthly_income_trade_and_tax #1
		}
	}
	proportion_for_payee = {
		add = {
			desc = "roman_tribute"
			value = 1
		}
	}
}
