	level = 4
	required_locations_fraction = 0.75
	rule = historical
	
	potential = {
		OR = {
			tag = c:BYZ

				owns = location:rome
				NOT = {
					tag = c:PAP
				}
			}
		}
	}
	
	allow = {
		owns = location:rome

		country_rank = country_rank:rank_empire

		}
	}
	
	name = ROM
