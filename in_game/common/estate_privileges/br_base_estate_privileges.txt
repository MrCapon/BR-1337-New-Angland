REPLACE_OR_CREATE:clergy_military_orders = {
	estate = clergy_estate

	potential = {
		OR = {
			religion = religion:catholic
			religion = religion:orthodox
		}
		OR = {
			has_reform = government_reform:military_order_reform
			any_neighbor_country = { NOT = { religion.group = root.religion.group } }
			culture = { has_culture_group = culture_group:iberian_group }
		}
	}
	allow = {
		has_advance = castle_advance
	}
	can_revoke = {
	}
	
	country_modifier = {
		global_clergy_estate_power = 0.25
		nobles_estate_levy_size = 0.25 #Added 
		clergy_estate_levy_size = 1 #Added 
		expected_army_size_modifier = 0.5 #BR added
	}

	country_modifier = {
		potential_trigger = { NOT = { has_reform = government_reform:military_order_reform } }
		clergy_estate_allowed_leading_military = yes
		discipline = 0.025
	}

	country_modifier = {
		potential_trigger = { has_reform = government_reform:military_order_reform }
		discipline = 0.05
	}

	country_modifier = {
		potential_trigger = {
			giving_scripted_relation_of_type = relation_type:military_sponsorship
		}
		land_morale_modifier = 0.1
	}

	country_modifier = {
		potential_trigger = {
			receiving_scripted_relation_of_type = relation_type:military_sponsorship
		}
		army_maintenance_cost = -0.15
	}
}
