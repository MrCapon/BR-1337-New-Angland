change_roman_authority = {
	international_organization:roman_world ?= {
		custom_description = {
			text = change_roman_authority_text
			value = $value$
			change_variable = {
				name = roman_authority
				add = $value$
			}
			if = {
				limit = { has_variable = roman_authority }
				if = {
					limit = { var:roman_authority > 100 }
					set_variable = { name = roman_authority value = 100 }
				}
				if = {
					limit = { var:roman_authority < 0 }
					set_variable = { name = roman_authority value = 0 }
				}
			}
		}
	}
}
