#Domestikos of the West
roman_world_bestow_domestikos_west = {
	type = diplomacy
	category = CATEGORY_FRIENDLY_ACTIONS

	potential = {
		exists = international_organization:roman_world
		international_organization:roman_world = {
			leader_country ?= scope:actor
			"num_countries_with_special_status(special_status:domestikos_west)" < "max_countries_with_special_status(special_status:domestikos_west)"
		}
	}

	allow = {
	}
	
	select_trigger = {
		looking_for_a = country
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			is_member_of_international_organization = international_organization:roman_world
		}
        enabled = {
			international_organization:roman_world = {
				special_status_can_be_bestowed = {
					type = special_status:domestikos_west
					country = root
				}
			}
		}
	}

	effect = {
		international_organization:roman_world = {
			international_organization_add_special_status = {
				type = special_status:domestikos_west
				country = scope:recipient
			}
			international_organization_remove_special_status = {
				type = special_status:imperial_loyalist
				country = scope:recipient
			}
		}
	}

	ai_will_do = {
		value = -100
		add = {
			desc = "OUR_OPINION_TOOLTIP"
			value = "scope:actor.opinion(scope:recipient)"
		}
		add = {
			desc = "THEIR_OPINION_TOOLTIP"
			value = "scope:recipient.opinion(scope:actor)"
		}
		if = {
			limit = {
				scope:actor.culture = scope:recipient.culture
			}
			add = 50
		}
		else_if = {
			limit = {
				scope:actor = {
					any_accepted_culture = {
						this = scope:recipient.culture
					}
				}
			}
			add = 25
		}
	}

	accept = {
		if = {
			limit = { exists = scope:recipient }
			add = {
				desc = "THEIR_OPINION_TOOLTIP"
				value = "scope:recipient.opinion(scope:actor)"
				if = {
					limit = {
						scope:recipient = {
							"opinion(scope:actor)" > 0
						}
					}
					divide = 10
				}
				else = {
					divide = 5
				}
			}
			add = {
				scope:actor = {
					value = modifier:diplomatic_reputation
					multiply = 3
				}
			}
			add = {
				scope:recipient = {
					value = "trust(scope:actor)"
					divide = 5
				}
			}
		}
	}
}
