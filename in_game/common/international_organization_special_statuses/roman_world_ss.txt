br_emperor = {
	max_countries = 1
	priority = 15
	can_bestow_trigger = {
		is_leader_of_international_organization = scope:recipient
	}

	#modifier = {
	#	monthly_prestige = 0.1
	#}
	auto_bestowal_trigger = {
		always = yes
	}
	auto_rescind_trigger = {
		NOT = { is_leader_of_international_organization = scope:recipient }
	}
	leader = yes
	map_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
	special_status_power = {
		value = 100
	}
}

domestikos_west = {
	max_countries = 1
	priority = 10

	modifier = {
		loyalty_to_overlord = 10
		monthly_legitimacy = 0.1
		army_maintenance_cost = -0.25
		expected_army_size = 5
	}

	leader_modifier = {
		court_spending_cost = 0.02
		government_size = 1
		num_local_governors = 1
	}

	can_bestow_trigger = {
		capital = {
			region = region:balkan_region
		}
		has_ruler = yes
		has_regent = no
		government_type = government_type:monarchy
		ruler ?= {
			is_adult = yes
			scope:recipient.leader_country = {
				has_ruler = yes
				ruler != prev
			}
		}
		has_special_status_in_international_organization = {
			type = special_status:imperial_loyalist
			international_organization = scope:recipient
		}
		religion = scope:recipient.leader_country.religion
	}

	auto_bestowal_trigger = {
	}

	auto_rescind_trigger = {
		OR = {
			has_ruler = no
			has_regent = yes
			ruler ?= {
				is_adult = no
				scope:recipient.leader_country = {
					has_ruler = yes
					ruler = prev
				}
			}
			religion != scope:recipient.leader_country.religion
		}
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
	}

	special_status_power = {
		value = 50
	}

	map_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_TIER6_COLOR
}

imperial_loyalist = {
	priority = 5

	modifier = {
		loyalty_to_overlord = 10
	}

	leader_modifier = {
		monthly_prestige = 0.005
	}

	can_bestow_trigger = {
	}

	auto_bestowal_trigger = {
		NOR = {
			has_special_status_in_international_organization = {
				type = special_status:br_emperor
				international_organization = scope:recipient
			}
			has_special_status_in_international_organization = {
				type = special_status:domestikos_west
				international_organization = scope:recipient
			}
			has_special_status_in_international_organization = {
				type = special_status:domestikos_east
				international_organization = scope:recipient
			}
		}

		religion = scope:recipient.leader_country.religion

		OR = {
			overlord ?= { this = scope:recipient.leader_country }
			ruler_or_regent ?= { scope:recipient.leader_country.ruler ?= this }
		}
	}

	auto_rescind_trigger = {
		religion = scope:recipient.leader_country.religion
		OR = {
			has_special_status_in_international_organization = {
				type = special_status:br_emperor
				international_organization = scope:recipient
			}
			has_special_status_in_international_organization = {
				type = special_status:domestikos_west
				international_organization = scope:recipient
			}
			has_special_status_in_international_organization = {
				type = special_status:domestikos_east
				international_organization = scope:recipient
			}
			has_special_status_in_international_organization = {
				type = special_status:defiant_strategoi
				international_organization = scope:recipient
			}
			NOR = {
				overlord ?= { this = scope:recipient.leader_country }
				ruler_or_regent ?= { scope:recipient.leader_country.ruler ?= this }
			}
		}
	}
	
	on_bestowed_effect = {
	}
	
	on_rescinded_effect = {
	}

	special_status_power = {
		value = 10
	}

	map_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_TIER2_COLOR
}
